SELECT 
organ_peo.OID,organ_ment.MID,organ.NAME,organ_ment.MENTNAME 
FROM organ_peo 
LEFT JOIN organ 
	on organ_peo.OID=organ.OID
	LEFT JOIN organ_ment
		on organ_peo.MID=organ_ment.MID
		WHERE organ_peo.UID='';
